server {

    listen       443 ssl;
    server_name  demo6.asapmall.kr;

    client_max_body_size 256M;

    ssl_certificate     /etc/letsencrypt/live/demo6.asapmall.kr/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/demo6.asapmall.kr/privkey.pem;

    location / {
       proxy_pass http://demo6.asapmall.kr:3000;
       proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
       proxy_set_header X-Forwarded-Proto $scheme;
    }

    location = /50x.html {
        root   /usr/share/nginx/html;
    }
}
