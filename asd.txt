    listen       443 ssl;
    server_name  demo1.asapmall.kr;
    
    client_max_body_size 256M;
    
    ssl_certificate     /etc/letsencrypt/live/demo1.asapmall.kr/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/demo1.asapmall.kr/privkey.pem;

    location / {
       proxy_pass http://demo1.asapmall.kr:3000;

    }
 
    location = /50x.html {
        root   /usr/share/nginx/html;
    }
}
